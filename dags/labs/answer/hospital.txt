from airflow.decorators import dag, task
from datetime import datetime, timedelta
import random
from airflow.providers.clickhousedb.hooks.clickhouse import ClickHouseHook

# ✅ BUG 2: เพิ่ม retries และ retry_delay เรียบร้อย
default_args = {
    "owner": "hospital-data-team",
    "email": ["data-alert@hospital.th"],
    "email_on_failure": True,
    "retries": 3,
    "retry_delay": timedelta(minutes=5),
}

# ✅ BUG 3: ย้าย hardcoded credentials ไปเก็บใน Airflow Connection (clickhouse_conn) และเรียกใช้ผ่าน ClickHouseHook
CLICKHOUSE_CONN_ID = "clickhouse_conn"


def get_client():
    ch_hook = ClickHouseHook(clickhouse_conn_id=CLICKHOUSE_CONN_ID)
    client = ch_hook.get_client()
    client.query("CREATE DATABASE IF NOT EXISTS labs")
    client.query("""
        CREATE TABLE IF NOT EXISTS labs.daily_stats (
            stat_date    String,
            test_type    String,
            total_tests  Int64,
            abnormal_cnt Int64,
            created_at   DateTime DEFAULT now()
        ) ENGINE = MergeTree()
        ORDER BY (stat_date, test_type)
    """)
    return client


def generate_lab_results(date: str) -> list:
    random.seed(date)
    test_types = ["CBC", "Blood Sugar", "Liver Function", "Kidney Function", "Thyroid"]
    results = []
    for i in range(20):
        value = round(random.uniform(50, 200), 1)
        results.append({
            "patient_id": f"HN-{random.randint(10000, 99999)}",
            "test_type":  random.choice(test_types),
            "result":     "Normal" if value < 150 else "Abnormal",
            "value":      value,
            "unit":       "mg/dL",
            "test_date":  date,
        })
    return results


@dag(
    dag_id="lab3_hospital_pipeline",
    default_args=default_args,
    schedule="0 7 * * *",   # รันทุกวัน 07:00 (พร้อมก่อน 08:00)
    start_date=datetime.now() - timedelta(days=1),
    catchup=False,
    tags=["lab3", "debug", "clickhouse"],
    doc_md=__doc__,
)
def lab3_hospital_pipeline():

    @task
    def fetch_lab_results(ds=None):
        """ดึงผล lab จากระบบโรงพยาบาล"""
        print(f"[fetch] กำลังดึงผล lab วันที่ {ds}")
        results = generate_lab_results(ds)
        print(f"[fetch] ได้ {len(results)} ผลตรวจ")
        return results

    @task
    def aggregate_stats(results: list, ds=None):
        """รวบรวมสถิติรายวัน"""
        client = get_client()
        # ลบข้อมูลเก่าของวันนั้นก่อน เพื่อความ idempotent
        client.query(f"DELETE FROM labs.daily_stats WHERE stat_date = '{ds}'", settings={"mutations_sync": "1"})
        
        # นับแต่ละ test_type
        from collections import Counter
        type_count = Counter(r["test_type"] for r in results)
        abnormal   = Counter(r["test_type"] for r in results if r["result"] == "Abnormal")
        
        rows = [
            (ds, test_type, total, abnormal.get(test_type, 0))
            for test_type, total in type_count.items()
        ]
        
        client.insert(
            "labs.daily_stats",
            rows,
            column_names=["stat_date", "test_type", "total_tests", "abnormal_cnt"]
        )
        print(f"[aggregate] บันทึกสถิติ {len(type_count)} test types วัน {ds} ✅")

    @task
    def send_report(ds=None):
        """ส่งรายงานสรุปประจำวัน"""
        client = get_client()
        result = client.query(
            "SELECT test_type, total_tests, abnormal_cnt FROM labs.daily_stats WHERE stat_date = {ds:String} ORDER BY abnormal_cnt DESC",
            parameters={"ds": ds}
        )
        rows = result.result_rows

        if not rows:
            raise ValueError(f"ไม่มีข้อมูล daily_stats วัน {ds} — aggregate ยังไม่เสร็จ?")

        print(f"\n{'='*45}")
        print(f"Hospital Lab Report — {ds}")
        print(f"{'Test Type':<20} {'Total':>8} {'Abnormal':>10}")
        print(f"{'-'*45}")
        for test_type, total, abnormal in rows:
            flag = " ⚠️" if abnormal > 3 else ""
            print(f"{test_type:<20} {total:>8} {abnormal:>10}{flag}")
        print(f"{'='*45}")
        print("[send_report] ส่งรายงานไปยัง email แล้ว (mock) ✅\n")

    results = fetch_lab_results()

    # ✅ BUG 1: แก้ไข dependency ลำดับที่ถูกต้อง: fetch_lab_results >> aggregate_stats >> send_report
    stats = aggregate_stats(results)
    stats >> send_report()

lab3_hospital_pipeline()
